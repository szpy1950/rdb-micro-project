Table City {
  id SERIAL PK
  cityName VARCHAR(255) [not null]
}

Table TrainType {
  id SERIAL PK
  trainTypeName VARCHAR(255) [not null]
}

Table Station {
  id SERIAL PK
  cityId INT [not null]
  stationName VARCHAR(255) [not null]

  indexes {
    cityId
  }
}

Table Line {
  id SERIAL PK
  startStationId INT [not null]
  endStationId INT [not null]
}

Table Segment {
  id SERIAL PK
  fromStationId INT [not null]
  toStationId INT [not null]
  duration INT [not null]
  price DECIMAL(10, 2) [not null]

  indexes {
    fromStationId
    toStationId
    (fromStationId, toStationId)
  }
}

Table Path {
  id SERIAL PK
  lineId INT [not null]
  trainTypeId INT [not null]
}

Table PathSegment {
  pathId INT [not null]
  segmentId INT [not null]
  segmentIndex INT [not null]

  indexes {
    (pathId, segmentId, segmentIndex) [pk]
    (pathId, segmentIndex) [unique]
  }
}

Table Train {
  id SERIAL PK
  typeId INT [not null]
  capacity INT [not null]

  indexes {
    typeId
    (typeId, capacity)
  }
}

Table TrainPath {
  trainId INT [not null]
  pathId INT [not null]
  departure TIME [not null]

  indexes {
    (trainId, pathId, departure) [pk]
  }
}

Table Ticket {
  id SERIAL PK
  fromStationId INT [not null]
  toStationId INT [not null]
  customerId INT [not null]
  price DECIMAL(10, 2) [not null]
  validUntil TIMESTAMP [not null]

  indexes {
    customerId
    (customerId, validUntil)
  }
}

Table Customer {
  id SERIAL PK
  firstName VARCHAR(255) [not null]
  lastName VARCHAR(255) [not null]
  birthdate DATE [not null]
}
